using System;
using System.Collections.Generic;
using System.Threading.Tasks;
using NativeWebSocket;
using TMPro;
using UnityEngine;

[Serializable]
public class EditUnityIntent
{
    public string type;
    public string intent;
    public string target;
    public string color;
}

public class ArkWebSocketClient : MonoBehaviour
{
    [Header("Connection")]
    public string serverUrl = "ws://localhost:8000/ws";

    [Header("UI")]
    public TMP_InputField commandInput;
    public TMP_Text statusText;

    [Header("Model Root")]
    public Transform importedModelRoot;

    private WebSocket _ws;

    async void Start()
    {
        await Connect();
    }

    private async Task Connect()
    {
        _ws = new WebSocket(serverUrl);

        _ws.OnOpen += () =>
        {
            if (statusText != null) statusText.text = "Connected";
        };

        _ws.OnError += (e) =>
        {
            if (statusText != null) statusText.text = "Socket Error: " + e;
        };

        _ws.OnClose += (e) =>
        {
            if (statusText != null) statusText.text = "Disconnected";
        };

        _ws.OnMessage += (bytes) =>
        {
            string msg = System.Text.Encoding.UTF8.GetString(bytes);
            HandleIncomingJson(msg);
        };

        await _ws.Connect();
    }

    private void HandleIncomingJson(string json)
    {
        if (json.Contains("\"intent\":\"edit_unity\"") || json.Contains("\"type\":\"edit_unity\""))
        {
            var intent = JsonUtility.FromJson<EditUnityIntent>(json);
            ApplyEditUnityIntent(intent);
        }
    }

    private void ApplyEditUnityIntent(EditUnityIntent intent)
    {
        if (importedModelRoot == null || string.IsNullOrWhiteSpace(intent.target))
            return;

        Transform targetPart = FindChildRecursive(importedModelRoot, intent.target);
        if (targetPart == null)
            return;

        AttachMeshColliderIfMissing(targetPart.gameObject);

        if (!string.IsNullOrWhiteSpace(intent.color))
        {
            if (TryParseNamedColor(intent.color, out Color parsed))
            {
                var renderer = targetPart.GetComponent<Renderer>();
                if (renderer != null)
                {
                    renderer.material = new Material(renderer.material);
                    renderer.material.color = parsed;
                }
            }
        }
    }

    private Transform FindChildRecursive(Transform root, string targetName)
    {
        if (root.name.Equals(targetName, StringComparison.OrdinalIgnoreCase))
            return root;

        foreach (Transform child in root)
        {
            var found = FindChildRecursive(child, targetName);
            if (found != null)
                return found;
        }

        return null;
    }

    private void AttachMeshColliderIfMissing(GameObject go)
    {
        var filter = go.GetComponent<MeshFilter>();
        if (filter == null || filter.sharedMesh == null)
            return;

        var mc = go.GetComponent<MeshCollider>();
        if (mc == null)
            mc = go.AddComponent<MeshCollider>();

        mc.sharedMesh = filter.sharedMesh;
        mc.convex = false;
    }

    private bool TryParseNamedColor(string name, out Color color)
    {
        switch (name.Trim().ToLowerInvariant())
        {
            case "red": color = Color.red; return true;
            case "green": color = Color.green; return true;
            case "blue": color = Color.blue; return true;
            case "yellow": color = Color.yellow; return true;
            case "black": color = Color.black; return true;
            case "white": color = Color.white; return true;
            case "gray": color = Color.gray; return true;
            case "cyan": color = Color.cyan; return true;
            case "magenta": color = Color.magenta; return true;
            default:
                if (ColorUtility.TryParseHtmlString(name, out color))
                    return true;
                color = Color.white;
                return false;
        }
    }

    public async void SendGenerateCommand()
    {
        if (_ws == null || _ws.State != WebSocketState.Open || commandInput == null)
            return;

        string payload = "{\"type\":\"generate\",\"prompt\":\"" + EscapeJson(commandInput.text) + "\"}";
        await _ws.SendText(payload);
    }

    private string EscapeJson(string input)
    {
        return (input ?? string.Empty).Replace("\\", "\\\\").Replace("\"", "\\\"");
    }

    async void OnApplicationQuit()
    {
        if (_ws != null)
            await _ws.Close();
    }

    void Update()
    {
#if !UNITY_WEBGL || UNITY_EDITOR
        _ws?.DispatchMessageQueue();
#endif
    }
}

using UnityEngine;

public class ModelRotator : MonoBehaviour
{
    public Transform target;
    public float rotateSpeed = 60f;

    public void RotateLeft()
    {
        if (target != null)
            target.Rotate(Vector3.up, rotateSpeed * Time.deltaTime, Space.World);
    }

    public void RotateRight()
    {
        if (target != null)
            target.Rotate(Vector3.up, -rotateSpeed * Time.deltaTime, Space.World);
    }

    public void RotateUp()
    {
        if (target != null)
            target.Rotate(Vector3.right, rotateSpeed * Time.deltaTime, Space.World);
    }

    public void RotateDown()
    {
        if (target != null)
            target.Rotate(Vector3.right, -rotateSpeed * Time.deltaTime, Space.World);
    }
}
